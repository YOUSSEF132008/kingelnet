document.addEventListener("DOMContentLoaded", function () { const buttons = document.querySelectorAll(".product button"); buttons.forEach(button => { button.addEventListener("click", (event) => { const productElement = event.target.parentElement; const productName = productElement.getAttribute("data-name"); const productPrice = parseFloat(productElement.getAttribute("data-price")); let cart = JSON.parse(localStorage.getItem("cart")) || []; cart.push({ name: productName, price: productPrice }); localStorage.setItem("cart", JSON.stringify(cart)); alert(`${productName} أُضيف للسلة`); }); }); });
